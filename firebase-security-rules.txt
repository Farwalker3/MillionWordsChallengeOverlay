rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Stories collection
    match /stories/{storyId} {
      // Anyone can read approved stories
      allow read: if resource.data.status == 'approved';
      
      // Anyone can create pending stories with 50+ words
      allow create: if request.auth == null 
                    && request.resource.data.status == 'pending'
                    && request.resource.data.wordCount >= 50;
      
      // Authenticated users (admins) can read, update, and delete ANY story
      allow read, update, delete: if request.auth != null;
    }
    
    // Banned users list (admin only)
    match /bannedUsers/{userId} {
      allow read, write: if request.auth != null;
    }
    
    // Deny everything else
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
